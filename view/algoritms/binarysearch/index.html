              function BinarySearch(array, item) {
                let start = 0;
                let end = array.length;
                let middle; 
                let found = false;
                let position = -1;
                while (found === false 'and' start 'меньше или равно' end) {
                  count += 1;
                  middle = Math.floor((start + end) / 2);
                  if (array[middle] === item) {
                    found = true;
                    position = middle;
                    return position;
                  }
                  if (item 'меньше' array[middle]) {
                    end = middle - 1;
                  } else {
                    start = middle + 1;
                  }
                }
                return position;
              }
